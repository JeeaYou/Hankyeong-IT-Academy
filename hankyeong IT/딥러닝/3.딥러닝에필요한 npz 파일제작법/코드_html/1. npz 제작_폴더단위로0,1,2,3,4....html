x,y=[],[]
folderList=glob('./clothes_dataset/*')

for cnt,folder in enumerate(folderList):
    fileList=glob(folder + '/*.jpg')
    print('▶ Start:' + folder)
    for file in fileList:
        img=Image.open(file)
        img=img.resize((180,180))
        imgArr=np.array(img)
        x.append(imgArr)
        y.append(cnt)
